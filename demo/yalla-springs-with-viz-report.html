Composite spec (YAML)

name: yalla-springs-with-viz
description: "yalla pair-wise ABM demo with a wired Visualization Step. Each update produces a Plotly HTML figure of the running scalar trajectories, written to the 'viz_html' store."

requires:
  processes: [YallaProcess, YallaSummaryPlots, RAMEmitter]

parameters:
  n_cells:
    type: int
    default: 40
    description: "Initial number of agents in the random sphere."
  force_kernel:
    type: string
    default: "spring"
    description: "Pair-wise force kernel."
  dt:
    type: float
    default: 0.02
    description: "Internal integrator sub-step size."
  interval:
    type: float
    default: 0.5
    description: "Process interval."
  seed:
    type: int
    default: 1
    description: "RNG seed for reproducibility."
  viz_title:
    type: string
    default: "yalla ABM summary"
    description: "Title displayed at the top of the Plotly figure."

state:
  yalla:
    _type: process
    address: "local:YallaProcess"
    config:
      n_cells: "${n_cells}"
      force_kernel: "${force_kernel}"
      dt: "${dt}"
      seed: "${seed}"
      init: "random_sphere"
      init_dist: 0.5
      L_0: 0.5
    inputs: {}
    outputs:
      n_cells: ["stores", "n_cells"]
      gyration_radius: ["stores", "gyration_radius"]
      mean_neighbor_distance: ["stores", "mean_neighbor_distance"]
      sorting_score: ["stores", "sorting_score"]
      type_radial_spread: ["stores", "type_radial_spread"]
      center_x: ["stores", "center_x"]
      center_y: ["stores", "center_y"]
      center_z: ["stores", "center_z"]
    interval: "${interval}"
  stores:
    n_cells: 0
    gyration_radius: 0.0
    mean_neighbor_distance: 0.0
    sorting_score: 0.0
    type_radial_spread: 0.0
    center_x: 0.0
    center_y: 0.0
    center_z: 0.0
    viz_html: ""
  viz:
    _type: step
    address: "local:YallaSummaryPlots"
    config:
      title: "${viz_title}"
    inputs:
      n_cells: ["stores", "n_cells"]
      gyration_radius: ["stores", "gyration_radius"]
      mean_neighbor_distance: ["stores", "mean_neighbor_distance"]
      sorting_score: ["stores", "sorting_score"]
      time: ["global_time"]
    outputs:
      html: ["stores", "viz_html"]
  emitter:
    _type: step
    address: "local:RAMEmitter"
    config:
      emit:
        n_cells: "integer"
        gyration_radius: "float"
        mean_neighbor_distance: "float"
        sorting_score: "float"
        type_radial_spread: "float"
        time: "float"
    inputs:
      n_cells: ["stores", "n_cells"]
      gyration_radius: ["stores", "gyration_radius"]
      mean_neighbor_distance: ["stores", "mean_neighbor_distance"]
      sorting_score: ["stores", "sorting_score"]
      type_radial_spread: ["stores", "type_radial_spread"]
      time: ["global_time"]
